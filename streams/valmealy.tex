\documentclass{standalone}

\usepackage{ tikz }
\usetikzlibrary{automata, positioning, arrows}

\begin{document}
    \begin{tikzpicture}[->, >=stealth, node distance=3cm, every state/.style={thick, fill=gray!10}, initial text=$ $, every edge/.append style={font=\tiny}]
        \node[state, accepting] (v) {$v$};
        \node[state, right of=v] (bot) {$\bot$};

        \draw (v) edge[above] node{$ () \,|\, v$} (bot)
              (bot) edge[loop right] node{$ () \,|\, \bot$} (bot)
        ;
    \end{tikzpicture}
\end{document}
